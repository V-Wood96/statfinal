run_balanced_model <- function(data, group_letter) { data <- data %>% mutate(Outcome = factor(ifelse(NTEE_MAJOR == group_letter, “Yes”, “No”)))

df_bal <- upSample( x = select(data, ASSET_s, INCOME_s, REV_s, RtoI_s), y = data$Outcome, yname = “Outcome” )

model <- glm(Outcome ~ ASSET_s + INCOME_s + REV_s + RtoI_s, data = df_bal, family = binomial)

pred <- ifelse(predict(model, type = “response”) > 0.5, “Yes”, “No”)

cm <- confusionMatrix( factor(pred, levels = c(“No”, “Yes”)), df_bal$Outcome )

list(model = model, confusion = cm) }
